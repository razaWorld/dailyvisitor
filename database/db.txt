-- 1. Drop the old table if it exists so we can start fresh
drop table if exists "user";

-- 2. Create the 'user' table with all your required fields
create table "user" (
  id bigint generated always as identity primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  name text not null,
  email text not null unique,
  password text not null, -- Stores password text
  role text default 'visitor', -- e.g., 'admin', 'user', 'visitor'
  lat numeric, -- Stores precise latitude coordinates
  long numeric, -- Stores precise longitude coordinates
  address text
);

-- 3. Turn on Row Level Security (RLS)
alter table "user" enable row level security;

-- 4. Create a policy that allows anyone to read the rows
create policy "Allow public read access"
on "user" for select
using (true);
